Definition mode_xor (m1 m2 : Mode) : Mode :=
  match m1 with
  | Pos ⇒ match m2 with
           | Pos ⇒ Pos
           | Neg ⇒ Neg
           end
  | Neg ⇒ match m2 with
           | Pos ⇒ Neg
           | Neg ⇒ Pos
           end
  end.

Lemma xor_prop_1: ∀ m,
    (mode_xor Pos m) = m.
Proof with auto.
  intros.
  destruct m...
Qed.

Lemma xor_prop_2: ∀ m,
    (mode_xor m Pos) = m.
Proof with auto.
  intros.
  destruct m...
Qed.

Lemma xor_prop_3: ∀ m1 m2,
    (mode_xor m1 m2) = mode_xor (flip m1) (flip m2).
Proof with auto.
  intros.
  destruct m1;destruct m2...
Qed.

Lemma xor_prop_4: ∀ m1 m2,
    flip (mode_xor m1 m2) = mode_xor (flip m1) ( m2).
Proof with auto.
  intros.
  destruct m1;destruct m2...
Qed.

Inductive typePairR : typ → typ → Prop :=
| tp_nat:
    typePairR typ_nat typ_nat
| tp_top: ∀ A ,
    type A →
    typePairR A typ_top
| tp_top_flip: ∀ A ,
    type A →
    typePairR typ_top A
| tp_fvar_x: ∀ X,
    typePairR (typ_fvar X) (typ_fvar X)
| tp_arrow: ∀ A1 A2 B1 B2,
    typePairR B1 A1 →
    typePairR A2 B2 →
    typePairR (typ_arrow A1 A2) (typ_arrow B1 B2)
| tp_rec: ∀ A B L,
    (∀ X, X \notin L →
               typePairR (open_tt A X) (open_tt B X)) →
    typePairR (typ_mu A) (typ_mu B).
